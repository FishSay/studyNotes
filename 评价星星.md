```javascript
// 评价星星
var stars = $api.domAll('.evaluateStar>.star-item');

for (var i = 0; i < stars.length; i++) {
    $api.addEvt(stars[i], 'click', function() {
        var starIndex = $api.text(this);

        if ($api.hasCls(this, 'star-fill')) {
            putOutStar(0, starIndex);
        } else {
            lightUpStar(0, starIndex);
        }
    });
}

var serviceStar1 = $api.domAll('.serviceStar1>.star-item');

for (var i = 0; i < serviceStar1.length; i++) {
    $api.addEvt(serviceStar1[i], 'click', function() {
        var starIndex = $api.text(this);

        if ($api.hasCls(this, 'star-fill')) {
            putOutStar(1, starIndex);
        } else {
            lightUpStar(1, starIndex);
        }
    });
}

var serviceStar2 = $api.domAll('.serviceStar2>.star-item');

for (var i = 0; i < serviceStar2.length; i++) {
    $api.addEvt(serviceStar2[i], 'click', function() {
        var starIndex = $api.text(this);

        if ($api.hasCls(this, 'star-fill')) {
            putOutStar(2, starIndex);
        } else {
            lightUpStar(2, starIndex);
        }
    });
}

var serviceStar3 = $api.domAll('.serviceStar3>.star-item');

for (var i = 0; i < serviceStar3.length; i++) {
    $api.addEvt(serviceStar3[i], 'click', function() {
        var starIndex = $api.text(this);

        if ($api.hasCls(this, 'star-fill')) {
            putOutStar(3, starIndex);
        } else {
            lightUpStar(3, starIndex);
        }
    });
}
// 点亮星星
function lightUpStar(what, number) {
    if (what == 0) {
        for (var j = 0; j < number; j++) {
            $api.addCls(stars[j], 'star-fill');
        }
    }
    if (what == 1) {
        for (var j = 0; j < number; j++) {
            $api.addCls(serviceStar1[j], 'star-fill');
        }
    }
    if (what == 2) {
        for (var j = 0; j < number; j++) {
            $api.addCls(serviceStar2[j], 'star-fill');
        }
    }
    if (what == 3) {
        for (var j = 0; j < number; j++) {
            $api.addCls(serviceStar3[j], 'star-fill');
        }
    }

}

//熄灭星星
function putOutStar(what, number) {
    if (what == 0) {
        for (var k = number; k < stars.length; k++) {
            $api.removeCls(stars[k], 'star-fill');
        }
    }
    if (what == 1) {
        for (var k = number; k < serviceStar1.length; k++) {
            $api.removeCls(serviceStar1[k], 'star-fill');
        }
    }
    if (what == 2) {
        for (var k = number; k < serviceStar2.length; k++) {
            $api.removeCls(serviceStar2[k], 'star-fill');
        }
    }
    if (what == 3) {
        for (var k = number; k < serviceStar3.length; k++) {
            $api.removeCls(serviceStar3[k], 'star-fill');
        }
    }

}


// 评价星星2
function serviceStar() {
    // 评价星星
    var serviceStar = $api.closest(event.target,'.serviceStar');
    var starsx = $api.domAll(serviceStar, '.star-item');
    var number = $api.text(event.target);
    // console.log(number);

    if ($api.hasCls(event.target, 'star-fill')) {
        for (var z = number; z < serviceStar3.length; z++) {
            $api.removeCls(starsx[z], 'star-fill');
        }
    } else {
        for (var x = 0; x < number; x++) {
            $api.addCls(starsx[x], 'star-fill');
        }
    }

}
```
